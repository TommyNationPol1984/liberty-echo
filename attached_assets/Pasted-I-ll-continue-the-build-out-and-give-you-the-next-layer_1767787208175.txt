@router.post("/create", response_model=VoiceResponse)

    category: str = Form("custom"),

🧩 2. api/services/voice_service.py
Extract embedding + save metadata

python
# api/services/voice_service.py

import os
import uuid
import json
from engine.models.speaker_encoder import SpeakerEncoder
from engine.utils.audio import load_wav
import numpy as np

VOICE_DIR = "voices/embeddings"
META_DIR = "voices"

encoder = SpeakerEncoder.load("checkpoints/speaker_encoder.pt")

async def create_voice_profile(name: str, category: str, audio_file):
    os.makedirs(VOICE_DIR, exist_ok=True)

    voice_id = str(uuid.uuid4())
    audio_path = f"{VOICE_DIR}/{voice_id}.wav"

    with open(audio_path, "wb") as f:
        f.write(await audio_file.read())

    embedding = encoder.embed_utterance(audio_path)
    np.save(f"{VOICE_DIR}/{voice_id}.npy", embedding)

    metadata = {
        "voice_id": voice_id,
        "name": name,
        "category": category,
        "embedding_path": f"{VOICE_DIR}/{voice_id}.npy"
    }

    with open(f"{META_DIR}/{voice_id}.json", "w") as f:
        json.dump(metadata, f, indent=4)

    return voice_id

# engine/inference/multilingual_tts.py

import soundfile as sf
import numpy as np
from engine.models.multilingual_model import MultilingualTTSModel
from engine.models.speaker_encoder import SpeakerEncoder

model = MultilingualTTSModel.load("checkpoints/multilingual_tts.pt")
encoder = SpeakerEncoder.load("checkpoints/speaker_encoder.pt")

def synthesize_multilingual(text, reference_audio, lang, out_path):
    speaker_embedding = encoder.embed_utterance(reference_audio)

    audio, sr = model.generate(
        text=text,
        speaker_embedding=speaker_embedding,
        language=lang,
        style="conversational"
    )

    sf.write(out_path, audio, sr)
    return out_path

# engine/models/multilingual_model.py

import numpy as np

class MultilingualTTSModel:
    def __init__(self, checkpoint_path: str):
        self.checkpoint_path = checkpoint_path

    @classmethod
    def load(cls, checkpoint_path: str):
        return cls(checkpoint_path)

    def generate(self, text, speaker_embedding, language="en", style="neutral"):
        sr = 22050
        audio = np.zeros(sr * 2, dtype=np.float32)  # 2 seconds silence placeholder
        return audio, sr

🧩 8. frontend/ starter (React)
frontend/pages/index.js
javascript
import { useState } from "react";

export default function Home() {
  const [text, setText] = useState("");
  const [voiceId, setVoiceId] = useState("");
  const [audioUrl, setAudioUrl] = useState("");

  async function generate() {
    const res = await fetch("http://localhost:8000/tts", {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({ text, voice_id: voiceId })
    });

    const job = await res.json();

    const interval = setInterval(async () => {
      const status = await fetch(`http://localhost:8000/tts/${job.job_id}`);
      const data = await status.json();

      if (data.status === "completed") {
        clearInterval(interval);
        setAudioUrl(data.audio_url);
      }
    }, 1000);
  }

  return (
    <div>
      <h1>Tommy Voice Engine</h1>
      <textarea value={text} onChange={e => setText(e.target.value)} />
      <input value={voiceId} onChange={e => setVoiceId(e.target.value)} placeholder="Voice ID" />
      <button onClick={generate}>Generate</button>
      {audioUrl && <audio controls src={audioUrl}></audio>}
    </div>
  );
}

🧠 5. Create workers/worker.py
This file continuously polls the job queue.

python
# workers/worker.py

import time
from api.services.job_queue import get_job
from workers.tasks.tts_task import run_tts_job

print("Worker started...")

while True:
    for job_id, job in list(get_job("*").items()):
        if job["status"] == "queued":
            print(f"Processing job {job_id}")
            run_tts_job(job_id, job["payload"])
    time.sleep(1)
(You can replace this with Celery or Redis later.)
